QUERY 1:

SELECT DISTINCT u.id, u.username
FROM users AS u
JOIN posts AS p
ON p.user_id = u.id
JOIN media_post AS mp
ON mp.post_id = p.id
JOIN medias AS m
ON m.id = mp.media_id
WHERE type = 'video'

QUERY 2:

SELECT p.id,p.title
FROM posts AS p
LEFT JOIN likes AS l
ON p.id = l.post_id
WHERE l.post_id IS NULL

QUERY 3:

SELECT COUNT(l.user_id) AS likes_number, p.id AS post_id, p.title
FROM posts AS p
LEFT JOIN likes AS l
ON p.id = l.post_id
GROUP BY p.id

QUERY 4:

SELECT u.id,u.username , COUNT(media_id) AS media_loaded
FROM users AS u
JOIN posts AS p
ON p.user_id = u.id
JOIN media_post AS mp
ON mp.post_id = p.id
GROUP BY u.id
